//3       6       9       12      15
//4       8       12      16      20

{
	int i,a,b;
	for(i=1;i<=4;i++)
	{
		for(a=1,b=i;a<=5;b=b+i,a++)
		printf("%d\t",b);
		printf("\n");
	}

}
