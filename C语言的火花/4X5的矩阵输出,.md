
## 输出下面4X5的矩阵。 
````c
1       2       3       4       5
2       4       6       8       10
3       6       9       12      15
4       8       12      16      20


一、个人解法
#include<stdio.h>
int main()
{
	int i,a,b;
	for(i=1;i<=4;i++)
	{
		for(a=1,b=i;a<=5;b=b+i,a++)
		printf("%d\t",b);
		printf("\n");
	}
	
	
	
	return 0;
} 



二、书上解法
#include<stdio.h>
int main()
{
	int i,j,n=0;
	for(i=1;i<=4;i++)
		for(j=1;j<=5;j++,n++)
		{	if(n%5==0)
			printf("\n");
			printf("%d\t",i*j);
		}
	printf("\n");
	return 0;
}
