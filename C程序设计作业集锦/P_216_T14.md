//一个学生的平均分的自定义函数 
float average(float s[10][5])
{	int i,j; 
	float snum,b;
	for(i=0;i<10;i++)
	{	snum=0;
		for(j=0;j<5;j++)
		{snum=snum+s[i][j];	}
		b=snum/5;
		printf("第%d同学的平均成绩为：%f。\n",i+1,b);
	}

}	

// 每门课的平均分自定义函数 
float average2(float s[10][5])
{	int i,j;
	float snum,b;
	for(i=0;i<5;i++)
	{	snum=0;
		for(j=0;j<10;j++)
		{snum=snum+s[j][i];	}
		b=snum/10;
		printf("第%d门课程的的平均成绩为：%f。\n",i+1,b);
	}

}	


//最高成绩对应的学生及其科目的自定义函数
int max(float s[10][5])
{	
	int i,j,x,y,n;
	n=s[0][0];
	for(i=0;i<10;i++)
	{	
		for(j=0;j<5;j++)
		{
			if(s[i][j]>=n) 
				{	n=s[i][j];
					x=i;
					y=j;
				}
				
		}

		
	}
printf("最高成绩为第%d个学生的第%d门成绩。\n",x+1,y+1);
}

//平均分方差的自定义函数
float FC(float s[10][5])
{	int	i,j;
	float snum;
	float b=0,fc1,fc2,c=0,Q;
	for(i=0;i<10;i++)
	{	snum=0;
		for(j=0;j<5;j++)
		{ snum=snum+s[i][j];	}
		fc1=(snum/5)*(snum/5);
		fc2=snum/5;
		b=b+fc1;	
		c=c+fc2;
	}
Q=(b/10)-(  (c/10)*(c/10)   );
printf("平均分方差为：%f。\n",Q);
