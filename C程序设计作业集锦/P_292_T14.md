//将n个数按输入时顺序的逆序排列，用函数实现
#include<stdio.h>

	int i,n;

void sort(char *p,int m)//将n个数逆序排列函数
{	
	int i,j;
	int temp;
	for(i=0,j=m-1;i<j;i++,j--)
	{
		temp=*(p+i);
		*(p+i)=*(p+j);
		*(p+j)=temp;
	}
}
